001    package com.thetransactioncompany.jsonrpc2.util;
002    
003    
004    import com.thetransactioncompany.jsonrpc2.JSONRPC2Error;

006    
007    /**
008     * The base abstract class for the JSON-RPC 2.0 parameter retrievers.
009     *
010     * @author Vladimir Dzhuvinov
011     * @version 1.31 (2012-12-22)
012     */
013    public abstract class ParamsRetriever {
014    
015            
016            /**
017             * Returns the parameter count.
018             *
019             * @return The parameters count.
020             */
021            public abstract int size();
022    
023    
024            /**
025             * Matches a string against an array of acceptable values.
026             *
027             * @param input       The string to match.
028             * @param enumStrings The acceptable string values. Must not be 
029             *                    {@code null}.
030             * @param ignoreCase  {@code true} for a case insensitive match.
031             *
032             * @return The matching string value, {@code null} if no match was
033             *         found.
034             */
035            protected static String getEnumStringMatch(final String input, 
036                                                       final String[] enumStrings, 
037                                                       final boolean ignoreCase) {
038            
039                    for (final String en: enumStrings) {
040                    
041                            if (ignoreCase) {
042                                    if (en.equalsIgnoreCase(input))
043                                            return en;
044                            }
045                            else {
046                                    if (en.equals(input))
047                                            return en;
048                            }
049                    }
050    
051                    return null;
052            }
053            
054            
055            /**
056             * Matches a string against an enumeration of acceptable values.
057             *
058             * @param input      The string to match.
059             * @param enumClass  The enumeration class specifying the acceptable 
060             *                   string values. Must not be {@code null}.
061             * @param ignoreCase {@code true} for a case insensitive match.
062             *
063             * @return The matching enumeration constant, {@code null} if no match
064             *         was found.
065             */
066            protected static <T extends Enum<T>> T getEnumStringMatch(final String input, 
067                                                                      final Class<T> enumClass, 
068                                                                      final boolean ignoreCase) {
069                    
070                    for (T en: enumClass.getEnumConstants()) {
071                    
072                            if (ignoreCase) {
073                                    if (en.toString().equalsIgnoreCase(input))
074                                            return en;
075                            }
076                            else {
077                                    if (en.toString().equals(input))
078                                            return en;
079                            }
080                    }
081                    
082                    return null;
083            }
084    }
