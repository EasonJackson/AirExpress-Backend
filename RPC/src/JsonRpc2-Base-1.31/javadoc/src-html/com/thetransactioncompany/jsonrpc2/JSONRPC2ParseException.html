001    package com.thetransactioncompany.jsonrpc2;
002    
003    
004    /** 
005     * Thrown to indicate an exception during the parsing of a JSON-RPC 2.0 
006     * message string.
007     *
008     * <p>The JSON-RPC 2.0 specification and user group forum can be found 
009     * <a href="http://groups.google.com/group/json-rpc">here</a>.
010     *
011     * @author Vladimir Dzhuvinov
012     * @version 1.31 (2012-11-28)
013     */
014    public class JSONRPC2ParseException extends Exception {
015            
016            
017            /**
018             * Serial version UID.
019             */
020            private static final long serialVersionUID = 3376608778436136410l;
021    
022    
023            /**
024             * Indicates a parse exception caused by a JSON message not conforming
025             * to the JSON-RPC 2.0 protocol.
026             */
027            public static int PROTOCOL = 0;
028            
029            
030            /**
031             * Indicates a parse exception caused by invalid JSON.
032             */
033            public static int JSON = 1;
034            
035            
036            /**
037             * The parse exception cause type. Default is {@link #PROTOCOL}.
038             */
039            private int causeType = PROTOCOL;
040            
041            
042            
043            /** 
044             * The string that could't be parsed.
045             */
046            private String unparsableString = null;
047            
048            
049            /** 
050             * Creates a new parse exception with the specified message. The cause 
051             * type is set to {@link #PROTOCOL}.
052             *
053             * @param message The exception message.
054             */
055            public JSONRPC2ParseException(final String message) {
056            
057                    super(message);
058            }
059            
060            
061            /**
062             * Creates a new parse exception with the specified message and the 
063             * original string that didn't parse. The cause type is set to
064             * {@link #PROTOCOL}.
065             *
066             * @param message          The exception message.
067             * @param unparsableString The unparsable string.
068             */
069            public JSONRPC2ParseException(final String message, final String unparsableString) {
070            
071                    super(message);
072                    this.unparsableString = unparsableString;
073            }
074            
075            
076            /**
077             * Creates a new parse exception with the specified message, cause type 
078             * and the original string that didn't parse.
079             *
080             * @param message          The exception message.
081             * @param causeType        The exception cause type, either 
082             *                         {@link #PROTOCOL} or {@link #JSON}.
083             * @param unparsableString The unparsable string.
084             */
085            public JSONRPC2ParseException(final String message, final int causeType, final String unparsableString) {
086            
087                    super(message);
088                    
089                    if (causeType != PROTOCOL && causeType != JSON)
090                            throw new IllegalArgumentException("Cause type must be either PROTOCOL or JSON");
091                    
092                    this.causeType = causeType;
093                    this.unparsableString = unparsableString;
094            }
095            
096            
097            /**
098             * Gets the parse exception cause type.
099             *
100             * @return The cause type, either {@link #PROTOCOL} or {@link #JSON}.
101             */
102            public int getCauseType() {
103            
104                    return causeType;
105            }
106            
107            
108            /**
109             * Gets original string that caused the parse exception (if specified).
110             *
111             * @return The string that didn't parse, {@code null} if none.
112             */
113            public String getUnparsableString() {
114            
115                    return unparsableString;
116            }
117    }
