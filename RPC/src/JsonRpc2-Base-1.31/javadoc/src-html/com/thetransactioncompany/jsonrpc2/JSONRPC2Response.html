007    import net.minidev.json.JSONObject;

010    /** 
011     * Represents a JSON-RPC 2.0 response.
012     *
013     * <p>A response is returned to the caller after a JSON-RPC 2.0 request has 
014     * been processed (notifications, however, don't produce a response). The
015     * response can take two different forms depending on the outcome:
016     *
017     * <ul>
018     *     <li>The request was successful. The corresponding response returns
019     *         a JSON object with the following information:
020     *         <ul>
021     *             <li>{@code result} The result, which can be of any JSON type
022     *                 - a number, a boolean value, a string, an array, an object 
023     *                 or null.
024     *             <li>{@code id} The request identifier which is echoed back back 
025     *                 to the caller.
026     *             <li>{@code jsonrpc} A string indicating the JSON-RPC protocol 
027     *                 version set to "2.0".
028     *         </ul>
029     *     <li>The request failed. The returned JSON object contains:
030     *         <ul>
031     *             <li>{@code error} An object with:
032     *                 <ul>
033     *                     <li>{@code code} An integer indicating the error type.
034     *                     <li>{@code message} A brief error messsage.
035     *                     <li>{@code data} Optional error data.
036     *                 </ul>
037     *             <li>{@code id} The request identifier. If it couldn't be 
038     *                 determined, e.g. due to a request parse error, the ID is
039     *                 set to {@code null}.
040     *             <li>{@code jsonrpc} A string indicating the JSON-RPC protocol 
041     *                 version set to "2.0".
042     *         </ul>
043     * </ul>
044     *
045     * <p>Here is an example JSON-RPC 2.0 response string where the request
046     * has succeeded:
047     *
048     * <pre>
049     * {  
050     *    "result"  : true,
051     *    "id"      : "req-002",
052     *    "jsonrpc" : "2.0"  
053     * }
054     * </pre>
055     *
056     *
057     * <p>And here is an example JSON-RPC 2.0 response string indicating a failure:
058     *
059     * <pre>
060     * {  
061     *    "error"   : { "code" : -32601, "message" : "Method not found" },
062     *    "id"      : "req-003",
063     *    "jsonrpc" : "2.0"
064     * }
065     * </pre>
066     *
067     * <p>A response object is obtained either by passing a valid JSON-RPC 2.0
068     * response string to the static {@link #parse} method or by invoking the
069     * appropriate constructor.
070     *
071     * <p>Here is how parsing is done:
072     * 
073     * <pre>
074     * String jsonString = "{\"result\":true,\"id\":\"req-002\",\"jsonrpc\":\"2.0\"}";
075     * 
076     * JSONRPC2Response response = null;
077     * 
078     * try {
079     *         response = JSONRPC2Response.parse(jsonString);
080     *
081     * } catch (JSONRPC2Exception e) {
082     *         // handle exception
083     * }
084     * </pre>
085     *
086     * <p>And here is how you can replicate the above example response strings:
087     *
088     * <pre>
089     * // success example
090     * JSONRPC2Response resp = new JSONRPC2Response(true, "req-002");
091     * System.out.println(resp);
092     * 
093     * // failure example
094     * JSONRPC2Error err = new JSONRPC2Error(-32601, "Method not found");
095     * resp = new JSONRPC2Response(err, "req-003");
096     * System.out.println(resp);
097     * 
098     * </pre>
099     *
100     * <p id="map">The mapping between JSON and Java entities (as defined by the 
101     * underlying JSON Smart library): 
102     *
103     * <pre>
104     *     true|false  <--->  java.lang.Boolean
105     *     number      <--->  java.lang.Number
106     *     string      <--->  java.lang.String
107     *     array       <--->  java.util.List
108     *     object      <--->  java.util.Map
109     *     null        <--->  null
110     * </pre>
111     *
112     * <p>The JSON-RPC 2.0 specification and user group forum can be found 
113     * <a href="http://groups.google.com/group/json-rpc">here</a>.
114     *
115     * @author Vladimir Dzhuvinov
116     * @version 1.31 (2012-11-11)
117     */
118    public class JSONRPC2Response extends JSONRPC2Message {
119            
120            
121            /** 
122             * The result. 
123             */
124            private Object result = null;
125            
126            
127            /** 
128             * The error object.
129             */
130            private JSONRPC2Error error = null;
131            
132            
133            /** 
134             * The echoed request identifier. 
135             */
136            private Object id = null;
137            
138            
139            /** 
140             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
141             *
142             * @param jsonString The JSON-RPC 2.0 response string, UTF-8 encoded.
143             *                   Must not be {@code null}.
144             *
145             * @return The corresponding JSON-RPC 2.0 response object.
146             *
147             * @throws JSONRPC2ParseException With detailed message if parsing 
148             *                                failed.
149             */
150            public static JSONRPC2Response parse(final String jsonString)
151                    throws JSONRPC2ParseException {
152            
153                    return parse(jsonString, false, false, false);
154            }
155            
156            
157            /** 
158             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
159             *
160             * @param jsonString    The JSON-RPC 2.0 response string, UTF-8 encoded.
161             *                      Must not be {@code null}.
162             * @param preserveOrder {@code true} to preserve the order of JSON 
163             *                      object members in results.
164             *
165             * @return The corresponding JSON-RPC 2.0 response object.
166             *
167             * @throws JSONRPC2ParseException With detailed message if parsing 
168             *                                failed.
169             */
170            public static JSONRPC2Response parse(final String jsonString, 
171                                                 final boolean preserveOrder)
172                    throws JSONRPC2ParseException {
173            
174                    return parse(jsonString, preserveOrder, false, false);
175            }
176            
177            
178            /** 
179             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
180             *
181             * @param jsonString    The JSON-RPC 2.0 response string, UTF-8 encoded.
182             *                      Must not be {@code null}.
183             * @param preserveOrder {@code true} to preserve the order of JSON 
184             *                      object members in results.
185             * @param ignoreVersion {@code true} to skip a check of the 
186             *                      {@code "jsonrpc":"2.0"} version attribute in the 
187             *                      JSON-RPC 2.0 message.
188             *
189             * @return The corresponding JSON-RPC 2.0 response object.
190             *
191             * @throws JSONRPC2ParseException With detailed message if the parsing 
192             *                                failed.
193             */
194            public static JSONRPC2Response parse(final String jsonString, 
195                                                 final boolean preserveOrder, 
196                                                 final boolean ignoreVersion)
197                    throws JSONRPC2ParseException {
198            
199                    return parse(jsonString, preserveOrder, ignoreVersion, false);
200            }
201            
202            
203            /** 
204             * Parses a JSON-RPC 2.0 response string. This method is thread-safe.
205             *
206             * @param jsonString            The JSON-RPC 2.0 response string, UTF-8 
207             *                              encoded. Must not be {@code null}.
208             * @param preserveOrder         {@code true} to preserve the order of  
209             *                              JSON object members in results.
210             * @param ignoreVersion         {@code true} to skip a check of the 
211             *                              {@code "jsonrpc":"2.0"} version 
212             *                              attribute in the JSON-RPC 2.0 message.
213             * @param parseNonStdAttributes {@code true} to parse non-standard
214             *                              attributes found in the JSON-RPC 2.0 
215             *                              message.
216             *
217             * @return The corresponding JSON-RPC 2.0 response object.
218             *
219             * @throws JSONRPC2ParseException With detailed message if the parsing 
220             *                                failed.
221             */
222            public static JSONRPC2Response parse(final String jsonString, 
223                                                 final boolean preserveOrder, 
224                                                 final boolean ignoreVersion,
225                                                 final boolean parseNonStdAttributes)
226                    throws JSONRPC2ParseException {
227            
228                    JSONRPC2Parser parser = new JSONRPC2Parser(preserveOrder, ignoreVersion, parseNonStdAttributes);
229                    
230                    return parser.parseJSONRPC2Response(jsonString);
231            }
232            
233            
234            /** 
235             * Creates a new JSON-RPC 2.0 response to a successful request.
236             *
237             * @param result The result. The value can <a href="#map">map</a> 
238             *               to any JSON type. May be {@code null}.
239             * @param id     The request identifier echoed back to the caller. May 
240             *               be {@code null} though not recommended.
241             */
242            public JSONRPC2Response(final Object result, final Object id) {
243            
244                    setResult(result);
245                    setID(id);
246            }
247            
248            
249            /** 
250             * Creates a new JSON-RPC 2.0 response to a successful request which
251             * result is {@code null}.
252             *
253             * @param id The request identifier echoed back to the caller. May be 
254             *           {@code null} though not recommended.
255             */
256            public JSONRPC2Response(final Object id) {
257            
258                    setResult(null);
259                    setID(id);
260            }
261            
262            
263            /** 
264             * Creates a new JSON-RPC 2.0 response to a failed request.
265             * 
266             * @param error A JSON-RPC 2.0 error instance indicating the
267             *              cause of the failure. Must not be {@code null}.
268             * @param id    The request identifier echoed back to the caller.
269             *              Pass a {@code null} if the request identifier couldn't
270             *              be determined (e.g. due to a parse error).
271             */
272            public JSONRPC2Response(final JSONRPC2Error error, final Object id) {
273            
274                    setError(error);
275                    setID(id);
276            }
277            
278            
279            /** 
280             * Indicates a successful JSON-RPC 2.0 request and sets the result. 
281             * Note that if the response was previously indicating failure this
282             * will turn it into a response indicating success. Any previously set
283             * error data will be invalidated.
284             *
285             * @param result The result. The value can <a href="#map">map</a> to 
286             *               any JSON type. May be {@code null}.
287             */
288            public void setResult(final Object result) {
289                    
290                    if (   result != null             &&
291                        ! (result instanceof Boolean) &&
292                        ! (result instanceof Number ) &&
293                        ! (result instanceof String ) &&
294                        ! (result instanceof List   ) &&
295                        ! (result instanceof Map    )    )
296                            throw new IllegalArgumentException("The result must map to a JSON type");
297                    
298                    // result and error are mutually exclusive
299                    this.result = result;
300                    this.error = null;
301            }       
302            
303            
304            /** 
305             * Gets the result of the request. The returned value has meaning
306             * only if the request was successful. Use the 
307             * {@link #getError getError} method to check this.
308             *
309             * @return The result.
310             */
311            public Object getResult() {
312                    
313                    return result;
314            }
315            
316            
317            /** 
318             * Indicates a failed JSON-RPC 2.0 request and sets the error details.
319             * Note that if the response was previously indicating success this
320             * will turn it into a response indicating failure. Any previously set 
321             * result data will be invalidated.
322             *
323             * @param error A JSON-RPC 2.0 error instance indicating the cause of 
324             *              the failure. Must not be {@code null}.
325             */
326            public void setError(final JSONRPC2Error error) {
327                    
328                    if (error == null)
329                            throw new IllegalArgumentException("The error object cannot be null");
330                    
331                    // result and error are mutually exclusive
332                    this.error = error;
333                    this.result = null;             
334            }
335            
336            
337            /** 
338             * Gets the error object indicating the cause of the request failure. 
339             * If a {@code null} is returned, the request succeeded and there was
340             * no error.
341             *
342             * @return A JSON-RPC 2.0 error object, {@code null} if the response
343             *         indicates success.
344             */
345            public JSONRPC2Error getError() {
346                    
347                    return error;
348            }
349            
350            
351            /**
352             * A convinience method to check if the response indicates success or
353             * failure of the request. Alternatively, you can use the 
354             * {@code #getError} method for this purpose.
355             *
356             * @return {@code true} if the request succeeded, {@code false} if
357             *         there was an error.
358             */
359            public boolean indicatesSuccess() {
360                    
361                    if (error == null)
362                            return true;
363                    else
364                            return false;
365            }
366            
367            
368            /**
369             * Sets the request identifier echoed back to the caller.
370             *
371             * @param id The value must <a href="#map">map</a> to a JSON scalar. 
372             *           Pass a {@code null} if the request identifier couldn't 
373             *           be determined (e.g. due to a parse error).
374             */
375            public void setID(final Object id) {
376                    
377                    if (   id != null             &&
378                        ! (id instanceof Boolean) &&
379                        ! (id instanceof Number ) &&
380                        ! (id instanceof String )    )
381                            throw new IllegalArgumentException("The request identifier must map to a JSON scalar");
382                    
383                    this.id = id;
384            }
385            
386            
387            /** 
388             * Gets the request identifier that is echoed back to the caller.
389             *
390             * @return The request identifier. If there was an error during the
391             *         the request retrieval (e.g. parse error) and the identifier 
392             *         couldn't be determined, the value will be {@code null}.
393             */
394             public Object getID() {
395             
396                    return id;
397            }
398            
399            
400            @Override
401            public JSONObject toJSONObject() {
402                    
403                    JSONObject out = new JSONObject();
404                    
405                    // Result and error are mutually exclusive
406                    if (error != null) {
407                            out.put("error", error.toJSONObject());
408                    }
409                    else {
410                            out.put("result", result);
411                    }
412                    
413                    out.put("id", id);
414                    
415                    out.put("jsonrpc", "2.0");
416                    
417                    
418                    Map <String,Object> nonStdAttributes = getNonStdAttributes();
419                    
420                    if (nonStdAttributes != null) {
421                    
422                            for (final Map.Entry<String,Object> attr: nonStdAttributes.entrySet())
423                                    out.put(attr.getKey(), attr.getValue());
424                    }
425                    
426                    return out;
427            }
