001    package com.thetransactioncompany.jsonrpc2;
002    
003    
004    import java.util.HashMap;
005    import java.util.List;
006    import java.util.Map;
007    
008    import net.minidev.json.JSONAware;
009    import net.minidev.json.JSONObject;
010    
011    
012    /**
013     * The base abstract class for JSON-RPC 2.0 requests, notifications and
014     * responses. Provides common methods for parsing (from JSON string) and
015     * serialisation (to JSON string) of these three message types.
016     *
017     * <p>Example parsing and serialisation back to JSON:
018     *
019     * <pre>
020     * String jsonString = "{\"method\":\"progressNotify\",\"params\":[\"75%\"],\"jsonrpc\":\"2.0\"}";
021     *
022     * JSONRPC2Message message = null;
023     *
024     * // parse
025     * try {
026     *        message = JSONRPC2Message.parse(jsonString);
027     * } catch (JSONRPC2ParseException e) {
028     *        // handle parse exception
029     * }
030     *
031     * if (message instanceof JSONRPC2Request)
032     *        System.out.println("The message is a request");
033     * else if (message instanceof JSONRPC2Notification)
034     *        System.out.println("The message is a notification");
035     * else if (message instanceof JSONRPC2Response)
036     *        System.out.println("The message is a response");
037     *
038     * // serialise back to JSON string
039     * System.out.println(message);
040     *
041     * </pre>
042     *
043     * <p id="map">The mapping between JSON and Java entities (as defined by the 
044     * underlying JSON Smart library): 
045     *
046     * <pre>
047     *     true|false  <--->  java.lang.Boolean
048     *     number      <--->  java.lang.Number
049     *     string      <--->  java.lang.String
050     *     array       <--->  java.util.List
051     *     object      <--->  java.util.Map
052     *     null        <--->  null
053     * </pre>
054     *
055     * <p>The JSON-RPC 2.0 specification and user group forum can be found 
056     * <a href="http://groups.google.com/group/json-rpc">here</a>.
057     * 
058     * @author Vladimir Dzhuvinov
059     * @version 1.31 (2012-11-30)
060     */
061    public abstract class JSONRPC2Message implements JSONAware {
062    
063    
064            /**
065             * Map of non-standard JSON-RPC 2.0 message attributes, {@code null} if
066             * none.
067             */
068            private Map <String,Object> nonStdAttributes = null;
069            
070    
071            /** 
072             * Provides common parsing of JSON-RPC 2.0 requests, notifications 
073             * and responses. Use this method if you don't know which type of 
074             * JSON-RPC message the input JSON string represents.
075             *
076             * <p>Batched requests / notifications are not supported.
077             *
078             * <p>This method is thread-safe.
079             *
080             * <p>If you are certain about the message type use the dedicated 
081             * {@link JSONRPC2Request#parse}, {@link JSONRPC2Notification#parse} 
082             * or {@link JSONRPC2Response#parse} methods. They are more efficient 
083             * and provide a more detailed parse error reporting.
084             *
085             * <p>The member order of parsed JSON objects will not be preserved 
086             * (for efficiency reasons) and the JSON-RPC 2.0 version field must be 
087             * set to "2.0". To change this behaviour check the optional {@link 
088             * #parse(String,boolean,boolean)} method.
089             *
090             * @param jsonString A JSON string representing a JSON-RPC 2.0 request, 
091             *                   notification or response, UTF-8 encoded. Must not
092             *                   be {@code null}.
093             *
094             * @return An instance of {@link JSONRPC2Request}, 
095             *         {@link JSONRPC2Notification} or {@link JSONRPC2Response}.
096             *
097             * @throws JSONRPC2ParseException With detailed message if parsing 
098             *                                failed.
099             */
100            public static JSONRPC2Message parse(final String jsonString)
101                    throws JSONRPC2ParseException {
102    
103                    return parse(jsonString, false, false);
104            }
105            
106            
107            /** 
108             * Provides common parsing of JSON-RPC 2.0 requests, notifications 
109             * and responses. Use this method if you don't know which type of 
110             * JSON-RPC message the input string represents.
111             *
112             * <p>Batched requests / notifications are not supported.
113             *
114             * <p>This method is thread-safe.
115             *
116             * <p>If you are certain about the message type use the dedicated 
117             * {@link JSONRPC2Request#parse}, {@link JSONRPC2Notification#parse} 
118             * or {@link JSONRPC2Response#parse} methods. They are more efficient 
119             * and provide a more detailed parse error reporting.
120             *
121             * @param jsonString    A JSON string representing a JSON-RPC 2.0 
122             *                      request, notification or response, UTF-8
123             *                      encoded. Must not be {@code null}.
124             * @param preserveOrder If {@code true} the member order of JSON objects
125             *                      in parameters and results must be preserved.
126             * @param ignoreVersion If {@code true} the {@code "jsonrpc":"2.0"}
127             *                      version field in the JSON-RPC 2.0 message will 
128             *                      not be checked.
129             *
130             * @return An instance of {@link JSONRPC2Request}, 
131             *         {@link JSONRPC2Notification} or {@link JSONRPC2Response}.
132             *
133             * @throws JSONRPC2ParseException With detailed message if parsing 
134             *                                failed.
135             */
136            public static JSONRPC2Message parse(final String jsonString, final boolean preserveOrder, final boolean ignoreVersion)
137                    throws JSONRPC2ParseException {
138                    
139                    JSONRPC2Parser parser = new JSONRPC2Parser(preserveOrder, ignoreVersion);
140                    
141                    return parser.parseJSONRPC2Message(jsonString);
142            }
143            
144            
145            /**
146             * Appends a non-standard attribute to this JSON-RPC 2.0 message. This is 
147             * done by adding a new member (key / value pair) to the top level JSON 
148             * object representing the message.
149             *
150             * <p>You may use this method to add meta and debugging attributes, 
151             * such as the request processing time, to a JSON-RPC 2.0 message.
152             *
153             * @param name  The attribute name. Must not conflict with the existing
154             *              "method", "id", "params", "result", "error" and "jsonrpc"
155             *              attributes reserved by the JSON-RPC 2.0 protocol, else 
156             *              an {@code IllegalArgumentException} will be thrown. Must
157             *              not be {@code null} either.
158             * @param value The attribute value. Must be of type String, boolean,
159             *              number, List, Map or null, else an
160             *              {@code IllegalArgumentException} will be thrown.
161             */
162            public void appendNonStdAttribute(final String name, final Object value) {
163            
164                    // Name check
165                    if (name == null          ||
166                        name.equals("method") ||
167                        name.equals("id")     ||
168                        name.equals("params") ||
169                        name.equals("result") ||
170                        name.equals("error")  ||
171                        name.equals("jsonrpc")   )
172            
173                            throw new IllegalArgumentException("Non-standard attribute name violation");
174            
175                    // Value check
176                    if ( value != null                &&
177                         ! (value instanceof Boolean) &&
178                         ! (value instanceof Number)  &&
179                         ! (value instanceof String)  &&
180                         ! (value instanceof List)    &&
181                         ! (value instanceof Map)        )
182                         
183                            throw new IllegalArgumentException("Illegal non-standard attribute value, must map to a valid JSON type");
184                    
185                    
186                    if (nonStdAttributes == null)
187                            nonStdAttributes = new HashMap<String,Object>();
188                    
189                    nonStdAttributes.put(name, value);
190            }
191            
192            
193            /**
194             * Retrieves a non-standard JSON-RPC 2.0 message attribute.
195             *
196             * @param name The name of the non-standard attribute to retrieve. Must
197             *             not be {@code null}.
198             *
199             * @return The value of the non-standard attribute (may also be 
200             *         {@code null}, {@code null} if not found.
201             */
202            public Object getNonStdAttribute(final String name) {
203            
204                    if (nonStdAttributes == null)
205                            return null;
206                    
207                    return nonStdAttributes.get(name);
208            }
209            
210            
211            /**
212             * Retrieves the non-standard JSON-RPC 2.0 message attributes.
213             *
214             * @return The non-standard attributes as a map, {@code null} if none.
215             */
216            public Map<String,Object> getNonStdAttributes() {
217            
218                    return nonStdAttributes;
219            }
220            
221            
222            /** 
223             * Returns a JSON object representing this JSON-RPC 2.0 message.
224             *
225             * @return The JSON object.
226             */
227            public abstract JSONObject toJSONObject();
228            
229            
230            /**
231             * Returns a JSON string representation of this JSON-RPC 2.0 message.
232             *
233             * @see #toString
234             *
235             * @return The JSON object string representing this JSON-RPC 2.0 
236             *         message.
237             */
238            @Override
239            public String toJSONString() {
240            
241                    return toString();
242            }
243            
244            
245            /** 
246             * Serialises this JSON-RPC 2.0 message to a JSON object string.
247             *
248             * @return The JSON object string representing this JSON-RPC 2.0 
249             *         message.
250             */
251            @Override
252            public String toString() {
253                    
254                    return toJSONObject().toString();
255            }
256    }
